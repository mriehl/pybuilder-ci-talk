Measure coverage (homemade)

#!/bin/bash
    coverage run py.test src
    COVERAGE=$(coverage report | tail -n 1 | ...
    echo "Coverage is $COVERAGE%"
    if [ $COVERAGE -lt 100 ]; then
      echo "Coverage sucks, breaking the build!" >
&1
      exit 1
    fi
